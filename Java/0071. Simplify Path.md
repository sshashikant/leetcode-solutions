class Solution {
    public String simplifyPath(String path) {
        String[] directories = path.split("/+");
        Stack<String> order = new Stack();
        
        for(String s : directories){
            if(order.size()>0 && "..".equals(s))
                order.pop();
            else if(!Arrays.asList("", ".", "..").contains(s))
                order.push(s);
        }
        return "/" + String.join("/", order);
    }
}
