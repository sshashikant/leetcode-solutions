class Solution {
    public ListNode swapNodes(ListNode head, int k) {
        ListNode count = head;
        int n = 0;
        while(count != null){
            n++;
            count = count.next;
        }
        
        ListNode first = head;
        for(int i=1; i<k; i++){
            first = first.next;
        }
        ListNode second = head;
        for(int i=1; i<=n-k; i++){
            second = second.next;
        }
        
        int temp = first.val;
        first.val = second.val;
        second.val = temp;
        
        return head;
    }
}
