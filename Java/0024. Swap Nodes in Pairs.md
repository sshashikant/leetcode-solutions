class Solution {
    public ListNode swapPairs(ListNode head) {
        ListNode dummy = new ListNode(0, head);
        ListNode prev = dummy;
        ListNode curr = head;
        
        while(curr!=null && curr.next!=null){
            //save ptrs
            ListNode nextPair = curr.next.next;
            ListNode second = curr.next;
            
            //reverse nodes
            second.next = curr;
            curr.next = nextPair;
            prev.next = second;
            
            //update ptrs
            prev = curr;
            curr = nextPair;
        }
        return dummy.next;
    }
}
