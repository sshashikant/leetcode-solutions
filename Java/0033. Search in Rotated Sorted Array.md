class Solution {
    public int search(int[] nums, int target) {
        int left = 0;
        int right = nums.length-1;
        
        while(left <= right) {
            if(nums[left] == target)
                return left;
            else if(nums[right] == target)
                return right;
            else if(target > nums[left])
                left++;
            else if(target < nums[right])
                right--;
            else
                break;
        }
        return -1;
    }
}
