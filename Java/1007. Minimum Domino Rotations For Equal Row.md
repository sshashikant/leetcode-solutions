class Solution {
    public int minDominoRotations(int[] tops, int[] bottoms) {
        int[] countTop = new int[7];
        int[] countBot = new int[7];
        int[] same = new int[7];
        
        int n = tops.length;
        for(int i=0; i<n; i++){
            countTop[tops[i]]++;
            countBot[bottoms[i]]++;
            
            if(tops[i]==bottoms[i])
                same[tops[i]]++;
        }
        
        for(int i=1; i<7; i++)
            if (countTop[i] + countBot[i] - same[i] == n)
                return n - Math.max(countTop[i], countBot[i]);
    return -1;
    }
}
