class Solution {
    public int findDuplicate(int[] nums) {
        int n = nums.length;
        int maxIndex = 0;
        int maxValue = 0;
        
        for(int i = 0; i < n; i++) {
            int id = nums[i]%n;
            nums[id] += n;
        }
        
        for(int i = 0; i < n; i++) {
            if(nums[i]>maxValue){
                maxValue = nums[i];
                maxIndex = i;
            }
            nums[i] %= n;
        }
        return maxIndex;
    }
}
